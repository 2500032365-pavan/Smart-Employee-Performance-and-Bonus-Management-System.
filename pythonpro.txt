employees = []while True:    print("\n--- MENU ---")    print("1. Add Employee")    print("2. View Employees")    print("3. Update Performance")    print("4. Calculate Bonus")    print("5. Exit")    choice = input("Enter your choice: ")    # Add Employee    if choice == '1':        emp_id = input("Enter ID: ")        name = input("Enter Name: ")        salary = float(input("Enter Salary: "))        score = int(input("Enter Performance Score: "))        employee = {            "id": emp_id,            "name": name,            "salary": salary,            "score": score        }        employees.append(employee)        print("Employee added successfully!")    # View Employees    elif choice == '2':        if len(employees) == 0:            print("No employees found.")        else:            for emp in employees:                print("\nID:", emp["id"])                print("Name:", emp["name"])                print("Salary:", emp["salary"])                print("Score:", emp["score"])    # Update Performance    elif choice == '3':        emp_id = input("Enter Employee ID: ")        found = False        for emp in employees:            if emp["id"] == emp_id:                new_score = int(input("Enter new performance score: "))                emp["score"] = new_score                print("Performance updated successfully!")                found = True                break        if not found:            print("Employee not found!")    # Calculate Bonus    elif choice == '4':        emp_id = input("Enter Employee ID: ")        found = False        for emp in employees:            if emp["id"] == emp_id:                score = emp["score"]                salary = emp["salary"]                if score >= 90:                    bonus = 0.20 * salary                elif score >= 75:                    bonus = 0.15 * salary                elif score >= 60:                    bonus = 0.10 * salary                else:                    bonus = 0.05 * salary                print(f"Bonus for {emp['name']} is: {bonus}")                found = True                break        if not found:            print("Employee not found!")    # Exit    elif choice == '5':        print("Thank You!")        break    else:        print("Invalid choice! Please try again.")
